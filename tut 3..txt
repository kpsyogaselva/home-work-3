#include<stdio.h>
#include<string.h>
int main()
{
   char string[100],word[20];
   int i,j,strleng,wrdleng,temp,count=0,check,Flag=0;
   gets(string);
   gets(word);
   strleng=strlen(string);
   wrdleng=strlen(word);
   for(i=0; i<strleng; i++)
   {
      temp=i;
      for(j=0;j<wrdleng;j++)
      {
         if(string[i]==word[j])
            i++;
      }
      check=i-temp;
      if(check==wrdleng)
         count++;
   }
   for(i=0;string[i]!='\0'; i++)
{
if(string[i]==word[0])
{
Flag=1;
for(j=0;word[j]!='\0';j++)
{
if(string[i+j]!=word[j])
{
Flag=0;
break;
}
}
}
if(Flag==1)
{
break;
}
}
if(Flag==0)
  {
  printf("\n Sorry!! We haven't found the Word '%s' ", word);
}
else
{
printf("\n We found '%s' at position %d ", word, i + 1);
}
printf("\nOccurrence = %d",count);
   return 0;
}
